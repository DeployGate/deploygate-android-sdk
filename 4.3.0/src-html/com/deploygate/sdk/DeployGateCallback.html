002package com.deploygate.sdk;
003
004
005/**
006 * A callback interface to receive DeployGate events. Implement this and pass to
007 * {@link DeployGate#registerCallback(DeployGateCallback, boolean)} to listen.
008 * 
009 * @author tnj
010 */
011public interface DeployGateCallback {
012    /**
013     * Callback to tell the finish of DeployGate service initialization
014     * procedure.
015     * 
016     * @param isServiceAvailable true if DeployGate service is available on the
017     *            device.
018     */
019    public void onInitialized(boolean isServiceAvailable);
020
021    /**
022     * Callback to tell the app status on the DeployGate has changed. This will
023     * also be called back just after {@link #onInitialized(boolean)},
024     * {@link DeployGate#refresh()} and
025     * {@link DeployGate#registerCallback(DeployGateCallback, boolean)} with
026     * true in second argument.
027     * 
028     * @param isManaged true if the app is known on DeployGate app
029     * @param isAuthorized true if the app is on current user's installed list
030     * @param loginUsername Current login username, returned only when
031     *            isAuthorized is true.
032     * @param isStopped Reserved.
033     */
034    public void onStatusChanged(boolean isManaged, boolean isAuthorized, String loginUsername,
035            boolean isStopped);
036
037    /**
038     * Callback to tell the new version is available.
039     * 
040     * @param revision revision of new version
041     * @param versionName user-defined version name of new version
042     * @param versionCode user-defined version code of new version
043     */
044    public void onUpdateAvailable(int revision, String versionName, int versionCode);
045}
