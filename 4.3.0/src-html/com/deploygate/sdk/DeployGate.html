105        public void onEvent(String action, Bundle extras) throws RemoteException {
106            if (DeployGateEvent.ACTION_INIT.equals(action)) {
107                onInitialized(
108                    extras.getBoolean(DeployGateEvent.EXTRA_IS_MANAGED, false),
109                    extras.getBoolean(DeployGateEvent.EXTRA_IS_AUTHORIZED, false),
110                    extras.getString(DeployGateEvent.EXTRA_LOGIN_USERNAME),
111                    extras.getString(DeployGateEvent.EXTRA_DISTRIBUTION_USER_NAME),
112                    extras.getBoolean(DeployGateEvent.EXTRA_IS_STOP_REQUESTED, false),
113                    extras.getString(DeployGateEvent.EXTRA_AUTHOR),
114                    extras.getInt(DeployGateEvent.EXTRA_CURRENT_REVISION, 0),
115                    extras.getString(DeployGateEvent.EXTRA_CURRENT_DISTRIBUTION_ID),
116                    extras.getString(DeployGateEvent.EXTRA_CURRENT_DISTRIBUTION_TITLE),
117                    extras.getInt(DeployGateEvent.EXTRA_DEPLOYGATE_VERSION_CODE,0)
118                );
119            }
120            else if (DeployGateEvent.ACTION_UPDATE_AVAILABLE.equals(action)) {
121                onUpdateArrived(extras.getInt(DeployGateEvent.EXTRA_SERIAL),
122                        extras.getString(DeployGateEvent.EXTRA_VERSION_NAME),
123                        extras.getInt(DeployGateEvent.EXTRA_VERSION_CODE),
124                        extras.getString(DeployGateEvent.EXTRA_SERIAL_MESSAGE)
125                    );
126            }
127            else if (DeployGateEvent.ACTION_ONESHOT_LOGCAT.equals(action)) {
128                onOneshotLogcat();
129            }
130            else if (DeployGateEvent.ACTION_ENABLE_LOGCAT.equals(action)) {
131                onEnableLogcat(true);
132            }
133            else if (DeployGateEvent.ACTION_DISABLE_LOGCAT.equals(action)) {
134                onEnableLogcat(false);
135            }
136        };

138        private void onInitialized(final boolean isManaged, final boolean isAuthorized,
139                                   final String loginUsername, final String distributionUserName,
140                                   final boolean isStopped, final String author,
141                                   int currentRevision, String distributionId, String distributionTitle, int deployGateVersionCode)
142                throws RemoteException {
143            Log.v(TAG, "DeployGate service initialized");
144            mAppIsManaged = isManaged;
145            mAppIsAuthorized = isAuthorized;
146            mAppIsStopRequested = isStopped;
147            mLoginUsername = loginUsername;
148            mDistributionUserName = distributionUserName;
149            mAuthor = author;
150            mDeployGateVersionCode = deployGateVersionCode;
151            mCurrentRevision = currentRevision;
152            mDistributionId = distributionId;
153            mDistributionTitle = distributionTitle;
154
155            mHandler.post(new Runnable() {
156                @Override
157                public void run() {
158                    for (DeployGateCallback callback : mCallbacks) {
159                        callback.onInitialized(true);
160                        callback.onStatusChanged(isManaged, isAuthorized, loginUsername, isStopped);
161                    }
162                }
163            });
164
165            mIsDeployGateAvailable = true;
166            mInitializedLatch.countDown();
167        }

169        private void onUpdateArrived(final int serial, final String versionName,
170                                     final int versionCode, final String message) throws RemoteException {
171            mAppUpdateAvailable = true;
172            mAppUpdateRevision = serial;
173            mAppUpdateVersionName = versionName;
174            mAppUpdateVersionCode = versionCode;
175            mAppUpdateMessage = message;
176
177            mHandler.post(new Runnable() {
178                @Override
179                public void run() {
180                    for (DeployGateCallback callback : mCallbacks) {
181                        callback.onUpdateAvailable(serial, versionName, versionCode);
182                    }
183                }
184            });
185        }

188    private void onOneshotLogcat() {
189        if (mLogcatThread == null || !mLogcatThread.isAlive()) {
190            mLogcatWorker = new LogCatTransportWorker(
191                    mApplicationContext.getPackageName(), mRemoteService, true);
192            mLogcatThread = new Thread(mLogcatWorker);
193            mLogcatThread.start();
194        }
195    }
196
197    private void onEnableLogcat(boolean isEnabled) {
198        if (mRemoteService == null)
199            return;
200
201        if (isEnabled) {
202            if (mLogcatThread == null || !mLogcatThread.isAlive()) {
203                mLogcatWorker = new LogCatTransportWorker(
204                        mApplicationContext.getPackageName(), mRemoteService, false);
205                mLogcatThread = new Thread(mLogcatWorker);
206                mLogcatThread.start();
207            }
208        } else {
209            if (mLogcatThread != null && mLogcatThread.isAlive()) {
210                mLogcatWorker.stop();
211                mLogcatThread.interrupt();
212            }
213        }
214    }

221                    callback.onInitialized(false);
222                    callback.onStatusChanged(false, false, null, false);

224            }
225        });

228    /**
229     * Do not instantiate directly. Call {@link #install(Application)} on your
230     * {@link Application#onCreate()} instead.
231     */
232    private DeployGate(Context applicationContext, String author, DeployGateCallback callback) {
233        mHandler = new Handler();
234        mApplicationContext = applicationContext;
235        mCallbacks = new HashSet<DeployGateCallback>();
236        mExpectedAuthor = author;
237
238        prepareBroadcastReceiver();
239
240        if (callback != null)
241            mCallbacks.add(callback);
242
243        mInitializedLatch = new CountDownLatch(1);
244        initService(true);
245    }
246
247    private boolean initService(boolean isBoot) {
248        if (isDeployGateAvailable()) {
249            Log.v(TAG, "DeployGate installation detected. Initializing.");
250            bindToService(isBoot);
251            return true;
252        } else {
253            Log.v(TAG, "DeployGate is not available on this device.");
254            mInitializedLatch.countDown();
255            mIsDeployGateAvailable = false;
256            callbackDeployGateUnavailable();
257            return false;
258        }
259    }
260
261    private boolean isDeployGateAvailable() {
262        String sig = getDeployGatePackageSignature();
263        if (sig == null)
264            return false;
265        for (String value : DEPLOYGATE_FINGERPRINTS)
266            if (value.equals(sig))
267                return true;
268        return false;
269    }
270
271    private void prepareBroadcastReceiver() {
272        IntentFilter filter = new IntentFilter(ACTION_DEPLOYGATE_STARTED);
273        mApplicationContext.registerReceiver(new BroadcastReceiver() {
274            @Override
275            public void onReceive(Context context, Intent intent) {
276                if (intent == null)
277                    return;
278                if (isDeployGateAvailable()) {
279                    bindToService(false);
280                }
281            }
282        }, filter);
283    }
284
285    private void bindToService(final boolean isBoot) {
286        Intent service = new Intent(IDeployGateSdkService.class.getName());
287        service.setPackage(DEPLOYGATE_PACKAGE);
288        mApplicationContext.bindService(service, new ServiceConnection() {
289            public void onServiceConnected(ComponentName name, IBinder service) {
290                Log.v(TAG, "DeployGate service connected");
291                mRemoteService = IDeployGateSdkService.Stub.asInterface(service);
292                requestServiceInit(isBoot);
293            }
294
295            @Override
296            public void onServiceDisconnected(ComponentName name) {
297                Log.v(TAG, "DeployGate service disconneced");
298                mRemoteService = null;
299            }
300        }, Context.BIND_AUTO_CREATE);
301    }
302
303    private void requestServiceInit(final boolean isBoot) {
304        Bundle args = new Bundle();
305        args.putBoolean(DeployGateEvent.EXTRA_IS_BOOT, isBoot);
306        args.putBoolean(DeployGateEvent.EXTRA_CAN_LOGCAT, canLogCat());
307        args.putString(DeployGateEvent.EXTRA_EXPECTED_AUTHOR, mExpectedAuthor);
308        args.putInt(DeployGateEvent.EXTRA_SDK_VERSION, SDK_VERSION);
309        try {
310            mRemoteService.init(mRemoteCallback, mApplicationContext.getPackageName(), args);
311        } catch (RemoteException e) {
312            Log.w(TAG, "DeployGate service failed to be initialized.");
313        }
314    }

316    private boolean canLogCat() {
317        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.JELLY_BEAN)
318            return true;
319        return mApplicationContext.getPackageManager().checkPermission(permission.READ_LOGS,
320                mApplicationContext.getPackageName()) == PackageManager.PERMISSION_GRANTED;
321    }

323    private String getDeployGatePackageSignature() {
324        PackageInfo info;
325        try {
326            info = mApplicationContext.getPackageManager().getPackageInfo(
327                    DEPLOYGATE_PACKAGE, PackageManager.GET_SIGNATURES);
328        } catch (NameNotFoundException e) {
329            return null;
330        }
331        if (info == null || info.signatures == null || info.signatures.length == 0)
332            return null;
333
334        MessageDigest md;
335        try {
336            md = MessageDigest.getInstance("SHA1");
337        } catch (NoSuchAlgorithmException e) {
338            Log.e(TAG, "SHA1 is not supported on this platform?", e);
339            return null;
340        }
341
342        byte[] digest = md.digest(info.signatures[0].toByteArray());
343        StringBuilder result = new StringBuilder(40);
344        for (int i = 0; i < digest.length; i++) {
345            result.append(Integer.toString((digest[i] & 0xff) + 0x100, 16).substring(1));
346        }
347        return result.toString();
348    }

498     * Install DeployGate on your application instance and register a callback
499     * listener. Call this method inside of your {@link Application#onCreate()}
500     * once.
501     * 
502     * @param app Application instance, typically just pass <em>this</em>.
503     * @param author author username of this app. Can be null.
504     * @param callback Callback interface to listen events. Can be null.
505     * @param forceApplyOnReleaseBuild if you want to keep DeployGate alive on
506     *            the release build, set this true.
507     * @since r2
508     */
509    public static void install(Application app, String author, DeployGateCallback callback,
510            boolean forceApplyOnReleaseBuild) {
511        if (sInstance != null) {
512            Log.w(TAG, "DeployGate.install was already called. Ignoring.");
513            return;
514        }
515
516        if (!forceApplyOnReleaseBuild && !isDebuggable(app.getApplicationContext()))
517            return;
518
519        Thread.setDefaultUncaughtExceptionHandler(new DeployGateUncaughtExceptionHandler(Thread
520                .getDefaultUncaughtExceptionHandler()));
521        sInstance = new DeployGate(app.getApplicationContext(), author, callback);
522    }

524    /**
525     * Request refreshing cached session values (e.g., isAuthorized, etc.) to
526     * the DeployGate service. Nothing happens if this called before
527     * {@link #install(Application)} or when refreshing is already in progress.
528     * Note that after calling this, {@link #isInitialized()} will changed to
529     * false immediately and any call to <code>is*()</code> will be blocked until
530     * refreshing get finished.
531     * 
532     * @since r1
533     */
534    public static void refresh() {
535        if (sInstance != null)
536            sInstance.refreshInternal();
537    }
538
539    private void refreshInternal() {
540        if (mInitializedLatch.getCount() == 0) {
541            mInitializedLatch = new CountDownLatch(1);
542            if (mRemoteService == null) {
543                initService(false);
544            } else {
545                requestServiceInit(false);
546            }
547        }
548    }

552     * {@link #unregisterCallback(DeployGateCallback)} when the callback is no
553     * longer needed (e.g., on destroying an activity.) If the listener has
554     * already in the callback list, just ignored.

562        if (sInstance == null)
563            return;

571        mCallbacks.add(listener);

835    private static class LogCatTransportWorker implements Runnable {
836        private final String mPackageName;
837        private final IDeployGateSdkService mService;
838        private Process mProcess;
839        private boolean mIsOneShot;
840
841        public LogCatTransportWorker(String packageName, IDeployGateSdkService service,
842                boolean isOneshot) {
843            mPackageName = packageName;
844            mService = service;
845            mIsOneShot = isOneshot;
846        }
847
848        @Override
849        public void run() {
850            mProcess = null;
851            ArrayList<String> logcatBuf = null;
852            BufferedReader bufferedReader = null;
853            ;
854            try {
855                LinkedList<String> commandLine = new LinkedList<String>();
856                commandLine.add("logcat");
857                logcatBuf = new ArrayList<String>();
858
859                int MAX_LINES = 500;
860                if (mIsOneShot) {
861                    commandLine.add("-d");
862                    if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.FROYO) {
863                        commandLine.add("-t");
864                        commandLine.add(String.valueOf(MAX_LINES));
865                    }
866                }
867                commandLine.add("-v");
868                commandLine.add("threadtime");
869                commandLine.add("*:V");
870
871                mProcess = Runtime.getRuntime().exec(
872                        commandLine.toArray(new String[commandLine.size()]));
873                bufferedReader = new BufferedReader(new InputStreamReader(
874                        mProcess.getInputStream()), 8192);
875
876                Log.v(TAG, "Start retrieving logcat");
877                String line;
878                while ((line = bufferedReader.readLine()) != null) {
879                    logcatBuf.add(line + "\n");
880                    if (mIsOneShot) {
881                        if (logcatBuf.size() > MAX_LINES)
882                            logcatBuf.remove(0);
883                    } else {
884                        if (!bufferedReader.ready()) {
885                            if (send(logcatBuf))
886                                logcatBuf.clear();
887                            else
888                                return;
889                        }
890                    }
891                }
892                
893                if (!logcatBuf.isEmpty())
894                    send(logcatBuf);
895                // EOF, stop it
896            } catch (IOException e) {
897                Log.d(TAG, "Logcat stopped: " + e.getMessage());
898            } finally {
899                if (bufferedReader != null) {
900                    try {
901                        bufferedReader.close();
902                    } catch (IOException e) {
903                        // ignored
904                    }
905                }
906                stop();
907            }
908        }
909
910        public void stop() {
911            if (mProcess != null)
912                mProcess.destroy();
913        }

915        private boolean send(ArrayList<String> logcatBuf) {
916            Bundle bundle = new Bundle();
917            bundle.putStringArrayList(DeployGateEvent.EXTRA_LOG, logcatBuf);
918            try {
919                mService.sendEvent(mPackageName, DeployGateEvent.ACTION_SEND_LOGCAT, bundle);
920            } catch (RemoteException e) {
921                return false;
922            }
923            return true;
924        }
925    }

931    void sendCrashReport(/* non-null */ Throwable ex) {
932        if (mRemoteService == null)
933            return;
934        
935        Bundle extras = new Bundle();
936        try {
937            if (getDeployGateVersionCode() >= SERIALIZED_EXCEPTION_SUPPORT_CLIENT_VERSION) {
938                Throwable rootCause = getRootCause(ex);
939                String msg = rootCause.getMessage();
940                extras.putString(DeployGateEvent.EXTRA_EXCEPTION_ROOT_CAUSE_CLASSNAME, rootCause.getClass().getName());
941                extras.putString(DeployGateEvent.EXTRA_EXCEPTION_ROOT_CAUSE_MESSAGE, msg != null ? msg : "");
942                extras.putString(DeployGateEvent.EXTRA_EXCEPTION_STACKTRACES, Log.getStackTraceString(ex));
943            } else {
944                extras.putSerializable(DeployGateEvent.EXTRA_EXCEPTION, ex);
945            }
946
947            mRemoteService.sendEvent(mApplicationContext.getPackageName(),
948                    DeployGateEvent.ACTION_SEND_CRASH_REPORT, extras);
949        } catch (RemoteException e) {
950            Log.w(TAG, "failed to send crash report: " + e.getMessage());
951        }
952    }
953
954    /* non-null */
955    private static Throwable getRootCause(/* non-null */ Throwable ex) {
956        Throwable cause = ex;
957        LinkedList<Throwable> throwables = new LinkedList<>();
958
959        while (cause != null && !throwables.contains(cause)) {
960            throwables.add(cause);
961            cause = cause.getCause();
962        }
963
964        return throwables.getLast();
965    }
