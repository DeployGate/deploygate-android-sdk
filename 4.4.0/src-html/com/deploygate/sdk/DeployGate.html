079        public void onEvent(
080                String action,
081                Bundle extras
082        ) throws RemoteException {
083            if (TextUtils.isEmpty(action)) {
084                return;
085            }
086
087            if (DeployGateEvent.ACTION_INIT.equals(action)) {
088                onInitialized(extras.getBoolean(DeployGateEvent.EXTRA_IS_MANAGED, false), extras.getBoolean(DeployGateEvent.EXTRA_IS_AUTHORIZED, false), extras.getString(DeployGateEvent.EXTRA_LOGIN_USERNAME), extras.getString(DeployGateEvent.EXTRA_DISTRIBUTION_USER_NAME), extras.getBoolean(DeployGateEvent.EXTRA_IS_STOP_REQUESTED, false), extras.getString(DeployGateEvent.EXTRA_AUTHOR), extras.getInt(DeployGateEvent.EXTRA_CURRENT_REVISION, 0), extras.getString(DeployGateEvent.EXTRA_CURRENT_DISTRIBUTION_ID), extras.getString(DeployGateEvent.EXTRA_CURRENT_DISTRIBUTION_TITLE));
089            } else if (DeployGateEvent.ACTION_UPDATE_AVAILABLE.equals(action)) {
090                onUpdateArrived(extras.getInt(DeployGateEvent.EXTRA_SERIAL), extras.getString(DeployGateEvent.EXTRA_VERSION_NAME), extras.getInt(DeployGateEvent.EXTRA_VERSION_CODE), extras.getString(DeployGateEvent.EXTRA_SERIAL_MESSAGE));
091            } else if (DeployGateEvent.ACTION_ONESHOT_LOGCAT.equals(action)) {
092                onOneshotLogcat();
093            } else if (DeployGateEvent.ACTION_ENABLE_LOGCAT.equals(action)) {
094                if (mDeployGateClient.isSupported(Compatibility.STREAMED_LOGCAT)) {
095                    String sessionKey = extras.getString(DeployGateEvent.EXTRA_LOGCAT_STREAM_SESSION_KEY);
096
097                    if (TextUtils.isEmpty(sessionKey)) {
098                        Logger.w("%s is missing in the extra", DeployGateEvent.EXTRA_LOGCAT_STREAM_SESSION_KEY);
099                        return;
100                    }
101
102                    onEnableStreamedLogcat(sessionKey);
103                } else {
104                    Logger.w("streamed logcat is not supported");
105                }
106            } else if (DeployGateEvent.ACTION_DISABLE_LOGCAT.equals(action)) {
107                if (mDeployGateClient.isSupported(Compatibility.STREAMED_LOGCAT)) {
108                    onDisableStreamedLogcat();
109                } else {
110                    Logger.w("streamed logcat is not supported");
111                }
112            }
113        }
114
115        private void onInitialized(
116                final boolean isManaged,
117                final boolean isAuthorized,
118                final String loginUsername,
119                final String distributionUserName,
120                final boolean isStopped,
121                final String author,
122                int currentRevision,
123                String distributionId,
124                String distributionTitle
125        ) throws RemoteException {
126            Log.v(TAG, "DeployGate service initialized");
127            mAppIsManaged = isManaged;
128            mAppIsAuthorized = isAuthorized;
129            mAppIsStopRequested = isStopped;
130            mLoginUsername = loginUsername;
131            mDistributionUserName = distributionUserName;
132            mAuthor = author;
133            mCurrentRevision = currentRevision;
134            mDistributionId = distributionId;
135            mDistributionTitle = distributionTitle;
136
137            mHandler.post(new Runnable() {
138                @Override
139                public void run() {
140                    for (DeployGateCallback callback : mCallbacks) {
141                        callback.onInitialized(true);
142                        callback.onStatusChanged(isManaged, isAuthorized, loginUsername, isStopped);
143                    }
144                }
145            });
146
147            mIsDeployGateAvailable = true;
148
149            mCustomLogInstructionSerializer.setDisabled(false);
150            mCustomLogInstructionSerializer.connect(mRemoteService);
151            mLogcatInstructionSerializer.setEnabled(true);
152            mLogcatInstructionSerializer.connect(mRemoteService);
153
154            mInitializedLatch.countDown();
155        }
156
157        private void onUpdateArrived(
158                final int serial,
159                final String versionName,
160                final int versionCode,
161                final String message
162        ) throws RemoteException {
163            mAppUpdateAvailable = true;
164            mAppUpdateRevision = serial;
165            mAppUpdateVersionName = versionName;
166            mAppUpdateVersionCode = versionCode;
167            mAppUpdateMessage = message;
168
169            mHandler.post(new Runnable() {
170                @Override
171                public void run() {
172                    for (DeployGateCallback callback : mCallbacks) {
173                        callback.onUpdateAvailable(serial, versionName, versionCode);
174                    }
175                }
176            });
177        }
178    };

214    /**
215     * Do not instantiate directly. Call {@link #install(Application)} on your
216     * {@link Application#onCreate()} instead.
217     */
218    private DeployGate(
219            Context applicationContext,
220            String author,
221            DeployGateCallback callback,
222            CustomLogConfiguration customLogConfiguration,
223            HostApp hostApp
224    ) {
225        mApplicationContext = applicationContext;
226        mDeployGateClient = new DeployGateClient(applicationContext, DEPLOYGATE_PACKAGE);
227        mHostApp = hostApp;
228        mHandler = new Handler();
229        mLogcatInstructionSerializer = mHostApp.canUseLogcat ? new LogcatInstructionSerializer(mHostApp.packageName) : ILogcatInstructionSerializer.NULL_INSTANCE;
230        mCustomLogInstructionSerializer = new CustomLogInstructionSerializer(mHostApp.packageName, customLogConfiguration);
231        mCallbacks = new HashSet<DeployGateCallback>();
232        mExpectedAuthor = author;
233
234        prepareBroadcastReceiver();
235
236        if (callback != null) {
237            mCallbacks.add(callback);
238        }
239
240        mInitializedLatch = new CountDownLatch(1);
241        initService(true);
242    }
243
244    private boolean initService(boolean isBoot) {
245        if (mDeployGateClient.isInstalled) {
246            Log.v(TAG, "DeployGate installation detected. Initializing.");
247            bindToService(isBoot);
248            return true;
249        } else {
250            Log.v(TAG, "DeployGate is not available on this device.");
251            mCustomLogInstructionSerializer.setDisabled(true);
252            mLogcatInstructionSerializer.setEnabled(false);
253            mInitializedLatch.countDown();
254            mIsDeployGateAvailable = false;
255            callbackDeployGateUnavailable();
256            return false;
257        }
258    }
259
260    private void prepareBroadcastReceiver() {
261        IntentFilter filter = new IntentFilter(ACTION_DEPLOYGATE_STARTED);
262        mApplicationContext.registerReceiver(new BroadcastReceiver() {
263            @Override
264            public void onReceive(
265                    Context context,
266                    Intent intent
267            ) {
268                if (intent == null) {
269                    return;
270                }
271                if (mDeployGateClient.isInstalled) {
272                    bindToService(false);
273                }
274            }
275        }, filter);
276    }
277
278    private void bindToService(final boolean isBoot) {
279        Intent service = new Intent(IDeployGateSdkService.class.getName());
280        service.setPackage(DEPLOYGATE_PACKAGE);
281        mApplicationContext.bindService(service, new ServiceConnection() {
282            public void onServiceConnected(
283                    ComponentName name,
284                    IBinder service
285            ) {
286                Log.v(TAG, "DeployGate service connected");
287                mRemoteService = IDeployGateSdkService.Stub.asInterface(service);
288                requestServiceInit(isBoot);
289            }
290
291            @Override
292            public void onServiceDisconnected(ComponentName name) {
293                Log.v(TAG, "DeployGate service disconneced");
294                mRemoteService = null;
295                mCustomLogInstructionSerializer.disconnect();
296                mLogcatInstructionSerializer.disconnect();
297            }
298        }, Context.BIND_AUTO_CREATE);
299    }
300
301    private void requestServiceInit(final boolean isBoot) {
302        Bundle args = new Bundle();
303        args.putBoolean(DeployGateEvent.EXTRA_IS_BOOT, isBoot);
304        args.putBoolean(DeployGateEvent.EXTRA_CAN_LOGCAT, mHostApp.canUseLogcat);
305        args.putString(DeployGateEvent.EXTRA_EXPECTED_AUTHOR, mExpectedAuthor);
306        args.putInt(DeployGateEvent.EXTRA_SDK_VERSION, mHostApp.sdkVersion);
307        try {
308            mRemoteService.init(mRemoteCallback, mHostApp.packageName, args);
309        } catch (RemoteException e) {
310            Log.w(TAG, "DeployGate service failed to be initialized.");
311        }
312    }
313
314    private void invokeAction(
315            String action,
316            Bundle extras
317    ) {
318        if (mRemoteService == null) {
319            return;
320        }
321        try {
322            mRemoteService.sendEvent(mHostApp.packageName, action, extras);
323        } catch (RemoteException e) {
324            Log.w(TAG, "failed to invoke " + action + " action: " + e.getMessage());
325        }
326    }

542    /**
543     * Install DeployGate on your application instance and register a callback
544     * listener. Call this method inside of your {@link Application#onCreate()}
545     * once.
546     *
547     * @param app
548     *         Application instance, typically just pass <em>this</em>.
549     * @param author
550     *         author username of this app. Can be null.
551     * @param callback
552     *         Callback interface to listen events. Can be null.
553     * @param forceApplyOnReleaseBuild
554     *         if you want to keep DeployGate alive on
555     *         the release build, set this true.
556     * @param customLogConfiguration
557     *         set a configuration for custom logging
558     *
559     * @since 4.4.0
560     */
561    public static void install(
562            Application app,
563            String author,
564            DeployGateCallback callback,
565            boolean forceApplyOnReleaseBuild,
566            CustomLogConfiguration customLogConfiguration
567    ) {
568        if (sInstance != null) {
569            Log.w(TAG, "DeployGate.install was already called. Ignoring.");
570            return;
571        }
572
573        HostApp hostApp = new HostApp(app.getApplicationContext());
574
575        if (!forceApplyOnReleaseBuild && !hostApp.debuggable) {
576            return;
577        }
578
579        Thread.setDefaultUncaughtExceptionHandler(new DeployGateUncaughtExceptionHandler(Thread.getDefaultUncaughtExceptionHandler()));
580        sInstance = new DeployGate(app.getApplicationContext(), author, callback, customLogConfiguration, hostApp);
581    }

583    /**
584     * Request refreshing cached session values (e.g., isAuthorized, etc.) to
585     * the DeployGate service. Nothing happens if this called before
586     * {@link #install(Application)} or when refreshing is already in progress.
587     * Note that after calling this, {@link #isInitialized()} will changed to
588     * false immediately and any call to <code>is*()</code> will be blocked until
589     * refreshing get finished.
590     *
591     * @since r1
592     */
593    public static void refresh() {
594        if (sInstance != null) {
595            sInstance.refreshInternal();
596        }
597    }
598
599    private void refreshInternal() {
600        if (mInitializedLatch.getCount() == 0) {
601            mInitializedLatch = new CountDownLatch(1);
602            if (mRemoteService == null) {
603                initService(false);
604            } else {
605                requestServiceInit(false);
606            }
607        }
608    }

610    /**
611     * Register a DeployGate event callback listener. Don't forget to call
612     * {@link #unregisterCallback(DeployGateCallback)} when the callback is no
613     * longer needed (e.g., on destroying an activity.) If the listener has
614     * already in the callback list, just ignored.
615     *
616     * @param listener
617     *         callback listener
618     * @param refreshImmediately
619     *         if you want to receive current states, set this
620     *         true.
621     *
622     * @since r1
623     */
624    public static void registerCallback(
625            DeployGateCallback listener,
626            boolean refreshImmediately
627    ) {
628        if (sInstance == null) {
629            return;
630        }
631        if (listener == null) {
632            return;
633        }
634
635        sInstance.registerCallbackInternal(listener, refreshImmediately);
636    }
637
638    private void registerCallbackInternal(
639            DeployGateCallback listener,
640            boolean callbackImmediately
641    ) {
642        mCallbacks.add(listener);
643        if (callbackImmediately) {
644            refresh();
645        }
646    }

917    void sendCrashReport(/* non-null */ Throwable ex) {
918        if (mRemoteService == null) {
919            return;
920        }
921
922        Bundle extras = new Bundle();
923        try {
924            if (mDeployGateClient.isSupported(Compatibility.SERIALIZED_EXCEPTION)) {
925                Throwable rootCause = getRootCause(ex);
926                String msg = rootCause.getMessage();
927                extras.putString(DeployGateEvent.EXTRA_EXCEPTION_ROOT_CAUSE_CLASSNAME, rootCause.getClass().getName());
928                extras.putString(DeployGateEvent.EXTRA_EXCEPTION_ROOT_CAUSE_MESSAGE, msg != null ? msg : "");
929                extras.putString(DeployGateEvent.EXTRA_EXCEPTION_STACKTRACES, Log.getStackTraceString(ex));
930            } else {
931                extras.putSerializable(DeployGateEvent.EXTRA_EXCEPTION, ex);
932            }
933
934            mRemoteService.sendEvent(mHostApp.packageName, DeployGateEvent.ACTION_SEND_CRASH_REPORT, extras);
935        } catch (RemoteException e) {
936            Log.w(TAG, "failed to send crash report: " + e.getMessage());
937        }
938    }

940    /* non-null */
941    private static Throwable getRootCause(/* non-null */ Throwable ex) {
942        Throwable cause = ex;
943        LinkedList<Throwable> throwables = new LinkedList<>();
944
945        while (cause != null && !throwables.contains(cause)) {
946            throwables.add(cause);
947            cause = cause.getCause();
948        }
949
950        return throwables.getLast();
951    }
