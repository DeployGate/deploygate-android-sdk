001package com.deploygate.sdk.internal;
002
003import android.util.Log;
004
005import java.util.Locale;
006
007public class Logger {
008    private static final String TAG = "DeployGateSDK";
009
010    public static void d(
011            String format,
012            Object... args
013    ) {
014        Log.d(TAG, String.format(Locale.US, format, args));
015    }
016
017    public static void i(
018            String format,
019            Object... args
020    ) {
021        Log.i(TAG, String.format(Locale.US, format, args));
022    }
023
024    public static void w(
025            String format,
026            Object... args
027    ) {
028        Log.w(TAG, String.format(Locale.US, format, args));
029    }
030
031    public static void w(
032            Throwable th,
033            String format,
034            Object... args
035    ) {
036        Log.w(TAG, String.format(Locale.US, format, args), th);
037    }
038
039    public static void e(
040            String format,
041            Object... args
042    ) {
043        Log.e(TAG, String.format(Locale.US, format, args));
044    }
045
046    public static void e(
047            Throwable th,
048            String format,
049            Object... args
050    ) {
051        Log.e(TAG, String.format(Locale.US, format, args), th);
052    }
053}
